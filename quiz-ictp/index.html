        async function init() {
            try {
                const response = await fetch('questions.json');
                if (!response.ok) throw new Error(`載入題目檔失敗：${response.status}`);
                questionsData = await response.json();
                progressText.innerText = `0/${questionsData.length}`;
            } catch (error) {
                quizContainer.innerHTML = `
                    

                        題庫讀取失敗：${error.message}
                    

                `;
                console.error(error);
                return;
            }

            questionsData.forEach((item) => {
                // 生成導覽按鈕
                const navBtn = document.createElement('button');
                navBtn.id = `nav-${item.id}`;
                navBtn.className = "h-6 w-full flex items-center justify-center bg-white border border-slate-200 rounded text-[10px] font-bold hover:border-blue-400 hover:text-blue-600 transition-all active:scale-95";
                navBtn.innerText = item.id;
                navBtn.onclick = () => scrollToQuestion(item.id);
                navGrid.appendChild(navBtn);

                // 生成題目卡片
                const qCard = document.createElement('div');
                qCard.id = `q-${item.id}`;
                qCard.className = "bg-white p-3 md:p-4 rounded shadow-sm border border-slate-200 transition-all hover:shadow-md";
                
                let codeHtml = item.code ? `
${escapeHtml(item.code)}
` : "";
                
                qCard.innerHTML = `
                    

                        Q${item.id}
                        
${escapeHtml(item.q)}

                    

                    ${codeHtml}
                    

                        ${item.o.map((opt, i) => `
                            
                        `).join('')}
                    

                    

                        

                        
正確答案：${String.fromCharCode(65 + item.a)}

                        ${item.note ? `
解析：${escapeHtml(item.note)}
` : ""}
                    

                `;
                quizContainer.appendChild(qCard);
            });
        }

        function scrollToQuestion(id) {
            const el = document.getElementById(`q-${id}`);
            el.scrollIntoView({ behavior: 'smooth', block: 'center' });
            el.classList.add('ring-2', 'ring-blue-500', 'border-blue-500');
            setTimeout(() => el.classList.remove('ring-2', 'ring-blue-500', 'border-blue-500'), 1500);

            toast.innerText = `已跳轉至 Q${id}`;
            toast.style.opacity = '1';
            setTimeout(() => toast.style.opacity = '0', 1000);
        }

        function checkAnswer(qId, selectedIdx, correctIdx) {
            const navBtn = document.getElementById(`nav-${qId}`);
            if (navBtn.classList.contains('answered-nav')) return;

            answeredCount++;
            navBtn.classList.add('answered-nav');

            const resultDiv = document.getElementById(`result-${qId}`);
            const feedback = document.getElementById(`feedback-${qId}`);

            for (let i = 0; i < 4; i++) {
                const btn = document.getElementById(`opt-${qId}-${i}`);
                btn.disabled = true;
                if (i === correctIdx) {
                    btn.classList.add('border-green-500', 'bg-green-50');
                    btn.querySelector('span').classList.add('border-green-500', 'text-green-600', 'bg-white');
                } else if (i === selectedIdx) {
                    btn.classList.add('border-red-400', 'bg-red-50');
                    btn.querySelector('span').classList.add('border-red-400', 'text-red-500', 'bg-white');
                } else {
                    btn.classList.add('opacity-40');
                }
            }

            resultDiv.classList.remove('hidden');
            if (selectedIdx === correctIdx) {
                correctCount++;
                feedback.innerText = "✓ 正確";
                feedback.className = "text-green-700 text-sm font-black mb-1";
                resultDiv.classList.add('correct-bg');
                navBtn.classList.add('correct-nav');
            } else {
                feedback.innerText = "✗ 錯誤";
                feedback.className = "text-red-700 text-sm font-black mb-1";
                resultDiv.classList.add('wrong-bg');
                navBtn.classList.add('wrong-nav');
            }

            const percent = (answeredCount / questionsData.length) * 100;
            progressBar.style.width = `${percent}%`;
            progressText.innerText = `${answeredCount}/${questionsData.length}`;
            
            const scorePercent = Math.round((correctCount / answeredCount) * 100);
            scoreText.innerText = `${scorePercent}% 正確`;
        }
